## 961. 在长度 2N 的数组中找出重复 N 次的元素

### 解法一
```
function repeatedNTimes(nums: number[]): number {
    let map = new Map<number, number>();
    for (const n of nums) {
        map.set(n, (map.get(n) || 0) + 1)
    }
    for (const [key, value] of map) {
        if (value === nums.length / 2) {
            return key;
        }
    }
    return 0;
};
```
