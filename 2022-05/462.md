## 462. 最少移动次数使数组元素相等 II

```
// 双循环
function minMoves2(nums: number[]): number {
    let result = Infinity;
    for (const n of nums) {
        let curr = 0;
        for (const c of nums) {
            curr += Math.abs(n - c);
        }
        result = Math.min(result, curr)
    }
    return result;
};
```
