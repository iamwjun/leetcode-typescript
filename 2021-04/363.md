## 368. 最大整除子集

解法一 不通过，题意是任意两个元素
```
function largestDivisibleSubset(nums: number[]): number[] {
    const n = nums.length;
    let i = n - 1, temp = new Array(n).fill(0);
    while (i >= 0) {
        if (temp[i]) {
            continue;
        }
        let res = [], j = 0, curr = nums[i];
        while (j <= i) {
            if (curr % nums[j] === 0 || nums[j] % curr === 0) {
                temp[j] = 1;
                res.push(nums[j]);
            }
            j++;
        }
        if (res.length >= i - 1) {
            return res;
        }
        i--;
    }
};
```
