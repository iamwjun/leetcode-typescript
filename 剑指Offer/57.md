## 剑指 Offer 55 - I. 二叉树的深度

#### 解法一 暴力
```
function findContinuousSequence(target: number): number[][] {
    let ans = [], i = 1;
    while (i <= target) {
        let temp = [], j = i, total = 0;
        while (j <= target) {
            total += j;
            temp.push(j);
            if (total === target && temp.length >= 2) {
                ans.push(temp);
                break;
            }
            if (total > target) {
                break;
            }
            j++;
        }
        i++;
    }
    return ans;
};
```
