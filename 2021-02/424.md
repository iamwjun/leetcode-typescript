## 424. 替换后的最长重复字符

解法一 暴力，没过示例 'BAAA'
```
function characterReplacement(s: string, k: number): number {
    let ans = 0, len = s.length;
    for (let i = 0; i < len; i++) {
        let step = 0;
        for (let j = i + 1; j < len; j++) {
            if (s[i] !== s[j]) step++;
            if (step > k) {
                ans = Math.max(ans, j - i);
                break
            } else {
                ans = Math.max(ans, j - i + 1);
            }
        }
    }
    return ans;
};
```
