## 896. 单调数列

解法一
```
function isMonotonic(A: number[]): boolean {
    const len = A.length;
    let ans = true;
    if (len === 2) return true;
    const gradient = A[len - 1] - A[0];
    for (let i = 1; i < len; i++) {
        if (
            (gradient < 0 && A[i] - A[i - 1] > 0) ||
            (gradient > 0 && A[i] - A[i - 1] < 0) ||
            (gradient === 0 && A[i] - A[i - 1] !== 0)
        ) {
            ans = false;
            break;
        }
    }
    return ans;
};
```
