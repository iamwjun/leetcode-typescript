## 665. 非递减数列

```
function checkPossibility(nums: number[]): boolean {
    let len = nums.length, ct = 0;
    for (let i = 0; i < len - 1; ++i) {
        let x = nums[i], y = nums[i + 1];
        if (x > y) {
            ct++;
            if (ct > 1) {
                return false;
            }
            if (nums[i - 1] > y) {
                nums[i + 1] = x;
            }
        }
    }
    return true;
};
```
